create table if not exists public.echo_noir_accounts (
  id text primary key,
  nickname text not null,
  total_kills integer not null default 0,
  total_xp integer not null default 0,
  level integer not null default 1,
  updated_at timestamptz not null default now()
);

alter table public.echo_noir_accounts enable row level security;

drop policy if exists "echo_noir_accounts_select_all" on public.echo_noir_accounts;
create policy "echo_noir_accounts_select_all"
on public.echo_noir_accounts for select
using (true);

drop policy if exists "echo_noir_accounts_upsert_all" on public.echo_noir_accounts;
create policy "echo_noir_accounts_upsert_all"
on public.echo_noir_accounts for insert
with check (true);

drop policy if exists "echo_noir_accounts_update_all" on public.echo_noir_accounts;
create policy "echo_noir_accounts_update_all"
on public.echo_noir_accounts for update
using (true)
with check (true);
